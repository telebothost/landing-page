JavaScript Basics for TBL Bots

This is a full crash course on JavaScript. By the end, you‚Äôll know enough to start coding bots in TBL.

1. Variables

Variables store data. Use let for changeable values and const for fixed values.

let name = "Alice";
const age = 21;

name = "Bob"; // allowed
// age = 22;  ‚ùå not allowed (const is fixed)

2. Data Types

JavaScript has different types of values:

  • String ‚Üí text ("Hello")
  • Number ‚Üí 42, 3.14
  • Boolean ‚Üí true or false
  • Array ‚Üí [1,2,3]
  • Object ‚Üí { key: "value" }
  • Undefined ‚Üí not set
  • Null ‚Üí empty on purpose
let str = "Hello"; 
let num = 100; 
let active = true; 
let items = ["apple", "banana"];
let user = { id: 1, name: "Soumya" };

3. Operators

Operators let you calculate and compare values.

let a = 10, b = 3;

console.log(a + b);  // 13
console.log(a - b);  // 7
console.log(a * b);  // 30
console.log(a / b);  // 3.33
console.log(a % b);  // 1

console.log(a > b);  // true
console.log(a === 10); // true

4. Strings

Strings are text. You can combine them and use functions:

let msg = "Hello";
let user = "Soumya";
console.log(msg + " " + user); 

console.log(msg.length);      // 5
console.log(user.toUpperCase()); // "SOUMYA"
console.log(user.includes("ou")); // true

5. Functions

Functions are reusable blocks of code.

function greet(name) {
  return "Hello " + name;
}

console.log(greet("Alice"));
console.log(greet("Bob"));

Modern style: arrow functions

const add = (x, y) => x + y;
console.log(add(5, 7));

6. Conditions

Decide what to do based on values:

let age = 18;

if(age >= 18) {
  console.log("Adult");
} else {
  console.log("Minor");
}

let fruit = "apple";
switch(fruit) {
  case "apple": console.log("Red fruit"); break;
  case "banana": console.log("Yellow fruit"); break;
  default: console.log("Unknown fruit");
}

7. Loops

Loops repeat actions:

// for loop
for(let i=1; i<=3; i++) {
  console.log("Step " + i);
}

// while loop
let n = 1;
while(n <= 3) {
  console.log("Number " + n);
  n++;
}

// for...of loop
let colors = ["red","blue","green"];
for(let c of colors) {
  console.log(c);
}

8. Arrays

Arrays are lists:

let fruits = ["apple","banana","mango"];
console.log(fruits[0]); // apple
fruits.push("orange");
console.log(fruits.length); // 4

// Loop through array
fruits.forEach(f => console.log("Fruit: " + f));

9. Objects

Objects store key‚Äìvalue pairs:

let user = {
  id: 1,
  name: "Soumya",
  premium: true
};

console.log(user.name); 
user.age = 20;
console.log(user);

10. ES6 Features

Modern JavaScript gives you powerful shortcuts:

// Template strings
let name = "Soumya";
console.log(`Hello ${name}`);

// Destructuring
let person = { id: 1, country: "India" };
let { id, country } = person;
console.log(id, country);

// Default params
function greet(name = "Guest") {
  return `Hi ${name}`;
}
console.log(greet());
console.log(greet("Alice"));

11. Callbacks

A callback is a function passed to another function:

function doTask(task, callback) {
  console.log("Doing: " + task);
  callback();
}

doTask("Download file", () => {
  console.log("Done!");
});

12. JavaScript Promises with Async/Await

Instead of using .then() chains, modern JavaScript uses async/await for cleaner async code. Below are the most important Promise methods you should know.

1. Promise.all()

Run many tasks in parallel and wait until all are finished. If any task fails ‚Üí the whole thing throws an error.

function task(name, time) {
  return new Promise(resolve => {
    setTimeout(() => resolve(name + " done"), time);
  });
}
// TBL doesn't support setTimeout()
async function runAll() {
  try {
    let results = await Promise.all([
      task("A", 1000),
      task("B", 2000),
      task("C", 500)
    ]);
    console.log(results); // [ "A done", "B done", "C done" ]
  } catch(err) {
    console.error("Error:", err.message);
  }
}
runAll();

2. Promise.allSettled()

Wait for all tasks, but never fail. Instead, you get the status of each task.

async function runAllSettled() {
  let results = await Promise.allSettled([
    Promise.resolve("OK"),
    Promise.reject("Oops")
  ]);
  console.log(results);
/*
[
  { status: "fulfilled", value: "OK" },
  { status: "rejected", reason: "Oops" }
]
*/
}
runAllSettled();

3. Promise.race()

Get the result of the first task that finishes (success or error).

async function runRace() {
  let result = await Promise.race([
    task("Fast", 500),
    task("Slow", 2000)
  ]);
  console.log(result); // "Fast done"
}
runRace();

4. Promise.any()

Get the result of the first successful task. If all fail ‚Üí throws AggregateError.

async function runAny() {
  try {
    let result = await Promise.any([
      Promise.reject("Fail1"),
      Promise.reject("Fail2"),
      Promise.resolve("Success üéâ")
    ]);
    console.log(result); // "Success üéâ"
  } catch(err) {
    console.error("All failed:", err.errors);
  }
}
runAny();

5. Simple Await Example

Every async function can await a promise directly.

async function load() {
  let data = await task("Loading data", 1000);
  console.log(data); // "Loading data done"
}
load();
